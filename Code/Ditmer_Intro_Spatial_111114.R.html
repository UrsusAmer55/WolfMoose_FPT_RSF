nobs<-nrow(used)

used$xt.pls1[1:nobs]<-c(used[2:nobs,"X_State"],NA)

used$xt.mn1[1:nobs]<-c(NA,used[1:(nobs-1),"X_State"])

used$estep<-with(used, sqrt((xt.pls1-X_State)^2+(yt.pls1-Y_State)^2))

# Create a unique identifier for each burst (we will set Tmax at 2 hours for determining bursts) 
#Will also want to keep track of first and last observations in a burst
 
Tmax<- 4
used$Burst = NA
BurstID = 1
used$Burst[1]<-1
used$firstb[1]<-1 # indicator of first obs. in a burst
used$lastb[1]<-0   # indicator of last obs. in a burst
for (row in 2:nobs){
  if (used$dt[row]<Tmax){
    used$Burst[row]=BurstID
    used$firstb[row]<-0
  } else {
    BurstID = BurstID+1
    used$Burst[row]=BurstID
    used$firstb[row]<-1
    used$lastb[row-1]<-1
  }  
}
used$lastb[nobs]<-1

#Delete observations for bursts containing a single location 
maxb<- max(used$Burst)
nb<-sapply(1:maxb, FUN=function(x){sum(used$Burst==x)})
sm.burst<-(1:maxb)[nb==1]
used2<-used[used$Burst%in%sm.burst!=T,]
used2$Burst<-as.factor(as.numeric(used2$Burst))

plot(density(used2$estep[used2$lastb!=T]), main="Empirical step length distribution")

# # Now, generate available points using an exponential distribution. 
# - Generate nav available points for each used point that is not a first point in the trajectory 
# - Assume turn angles are uniformly distributed (0, 360).  
# 
# Note; the first observations in a burst will not be considered in 
# the analysis, but will only be used to generate available points at the subsequent
# observation time.
# 
# -  used.f contains all used points that will be included in the anlysis
# -  used.tm1 contains all used pionts that will be conditioned on when determining
# available points. 


used.f<-used2[used2$firstb!=1,] # drops first observations...
nused<-nrow(used.f) # total number of used points in the analysis
used.tm1<-used2[used2$lastb!=1,] # use these data to generate the used points


# Make sure steps are large enough to encompass all empirical steps, so 
# will assume mean step length is double what we have observed (see Forester 2009)

mu.step<-2*mean(used.tm1$estep, na.rm=T)

#Again, can alter nav to generate more available points:used point 

nav<-3 #points per used point
steps<-rexp(nav*nused, 1/mu.step)
turns<-runif(nav*nused, 0,360)
avails<-data.frame(X_State=rep(NA,nav*nused))
avails$X_State<-rep(used.tm1$X_State,each=nav)+steps*cos(turns)  
avails$Y_State<-rep(used.tm1$Y_State,each=nav)+steps*sin(turns)
